86  int setRateNumeric(int rateNum) {
87  int diff = 99999;
88  int result = 0;
89  for (int j = 0; j < 8; j++) {
90  if (rate_num[j] == rateNum) {
91  AD_LOGD("-> %d", rate_num[j]);
92  i2s.rate = rate_code[j];
93  return rateNum;
94  } else {
95  int new_diff = abs(rate_code[j] - rateNum);
96  if (new_diff < diff) {
97  result = j;
98  diff = new_diff;
99  }
100  }
101  }
102  AD_LOGE("Sample Rate not supported: %d - using %d", rateNum,
103  rate_num[result]);
104  i2s.rate = rate_code[result];
105  return rate_num[result];
106  }
