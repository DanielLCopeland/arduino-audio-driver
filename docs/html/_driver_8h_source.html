96  int setRateNumeric(int rateNum) {
97  int diff = 99999;
98  int result = 0;
99  for (int j = 0; j < 8; j++) {
100  if (rate_num[j] == rateNum) {
101  AD_LOGD("-> %d", rate_num[j]);
102  i2s.rate = rate_code[j];
103  return rateNum;
104  } else {
105  int new_diff = abs(rate_code[j] - rateNum);
106  if (new_diff < diff) {
107  result = j;
108  diff = new_diff;
109  }
110  }
111  }
112  AD_LOGE("Sample Rate not supported: %d - using %d", rateNum,
113  rate_num[result]);
114  i2s.rate = rate_code[result];
115  return rate_num[result];
116  }
