288  virtual bool begin() {
289  // setup spi
290  bool result = true;
291  for (auto &tmp : spi) {
292  result &= tmp.begin();
293  }
294  // setup i2c
295  for (auto &tmp : i2c) {
296  result &= tmp.begin();
297  }
298  // setup pins
299  for (auto &tmp : pins) {
300  if (tmp.pin != -1) {
301  switch (tmp.pin_logic) {
302  case PinLogic::InputActiveHigh:
303  pinMode(tmp.pin, INPUT);
304  break;
305  case PinLogic::InputActiveLow:
306  pinMode(tmp.pin, INPUT_PULLUP);
307  break;
308  case PinLogic::Input:
309  pinMode(tmp.pin, INPUT);
310  break;
311  case PinLogic::Output:
312  pinMode(tmp.pin, OUTPUT);
313  break;
314  default:
315  // do nothing
316  break;
317  }
318  }
319  }
320  return result;
321  }

323  void end() {
324  // setup spi
325  for (auto &tmp : spi) {
326  tmp.end();
327  }
328  // setup i2c

331  }
