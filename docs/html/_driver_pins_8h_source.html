103  // setup chip select

105  pinMode(cs, OUTPUT);

118  p_spi->setMOSI(mosi);
119  p_spi->setSCLK(clk);
120  p_spi->setSSEL(cs);
121  p_spi->begin();
122 #elif defined(ESP32)
123  AD_LOGI("setting up SPI miso:%d,mosi:%d, clk:%d, cs:%d", miso, mosi,

127  AD_LOGW("setting up SPI w/o pins");

130  }
131  }
132  return true;
133  }

364  void setupPinMode() {
365  // setup pins
366  for (auto &tmp : pins) {
367  if (tmp.pin != -1) {
368  if (!hasConflict(tmp.pin)) {
369  AD_LOGD("pinMode %d", tmp.pin);
370  switch (tmp.pin_logic) {
371  case PinLogic::InputActiveHigh:
372  pinMode(tmp.pin, INPUT);
373  break;
374  case PinLogic::InputActiveLow:
375  pinMode(tmp.pin, INPUT_PULLUP);
376  break;
377  case PinLogic::Input:
378  pinMode(tmp.pin, INPUT);
379  break;
380  case PinLogic::Output:
381  pinMode(tmp.pin, OUTPUT);
382  break;
383  default:
384  // do nothing
385  break;
386  }
387  } else {
388  AD_LOGW("Pin '%d' not set up because of conflict", tmp.pin);
389  tmp.active = false;
390  }
391  }
392  }
