266  virtual bool begin() {
267  // setup spi
268  bool result = true;
269  for (auto &tmp : spi) {
270  result &= tmp.begin();
271  }
272  // setup i2c
273  for (auto &tmp : i2c) {
274  result &= tmp.begin();
275  }
276  // setup pins
277  for (auto &tmp : pins) {
278  if (tmp.pin != -1) {
279 
280  switch (tmp.function) {
281  case HEADPHONE_DETECT:
282  pinMode(tmp.pin, INPUT_PULLUP);
283  break;
284  case AUXIN_DETECT:
285  pinMode(tmp.pin, INPUT_PULLUP);
286  break;
287  case KEY:
288  pinMode(tmp.pin, INPUT);
289  break;
290  case LED:
291  pinMode(tmp.pin, OUTPUT);
292  break;
293  case PA:
294  pinMode(tmp.pin, OUTPUT);
295  digitalWrite(tmp.pin, HIGH);
296  break;
297  default:
298  // do nothing
299  break;
300  }
301  }
302  }
303  return result;
304  }

306  void end() {
307  // setup spi
308  for (auto &tmp : spi) {
309  tmp.end();
310  }
311  // setup i2c

314  }
